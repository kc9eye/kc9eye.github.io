class SearchStringFormater {

    public function __construct () {
        return $this;
    }

    public function formatSearchString ($string) {
        $string = trim($string);
        $pattern[0] = '/("|\')/i';
        $pattern[1] = '/\b (AND|&) \b/';
        $pattern[2] = '/\b (OR|\|) \b/';
        $pattern[3] = '/\b( +)\b/i';
        $pattern[4] = '/^(.*)$/i';
        $replacment[0] = ' ';
        $replacment[1] = ' & ';
        $replacment[2] = ' | ';
        $replacment[3] = ' | ';
        $replacment[4] = '$1';
        return preg_filter($pattern,$replacment,$string);
    }

}
