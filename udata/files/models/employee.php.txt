class Employee {

    protected $dbh;
    protected $storage;

    public function __construct (PDO $dbh, $storage) {
        $this->dbh = $dbh;
        $this->storage = $storage;
    }

    public function listCurrentEmployees (Array $filter = null) {
        try {
            $sql = 
            'SELECT employee.id as id, employee.start_date as start, employee.term_date as term, employee.status as status,
            profiles.first as first, profiles.middle as middle, profiles.last as last, profiles.other as other
            FROM employee
            INNER JOIN profiles
            ON employee.pid = profiles.id
            WHERE employee.term_date IS NULL';
            $pntr = $this->dbh->query($sql);
            return $pntr->fetchAll(PDO::FETCH_ASSOC);
        }
        catch (PDOException $e) {
            trigger_error($e->getMessage(),E_USER_WARNING);
            return false;
        }
    }
